获取验证码倒计时效果

					var validCode = true;
					$(document).on("click", ".ele", function() {
						if(validCode) {
							validCode = false;
							var time = 60;
							var _this = $(this);
							_this.addClass("disabled");
							var t = setInterval(function() {
								time--;
								_this.html(time + "秒");
								if(time == 0) {
									clearInterval(t);
									_this.html("重新获取");
									_this.removeClass("disabled");
									validCode = true;
								}
							}, 1000)
						}
					})
